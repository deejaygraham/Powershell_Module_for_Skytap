  10 function Set-Authorization ([string]$tokenfile='user_token', [string]$user, [string]$pwd) {
  11 <#
  12     .SYNOPSIS
  13       Creates authorization headers from file or parameters
  14     .SYNTAX
  15        Add-ConfigurationToProject EnvironmentId ProjectId
  16     .EXAMPLE
  17       Add-ConfigurationToProject 12345 54321
  18 #>
  19       if ($user) {    #use params instead of file 
  20           $username = $user
  21           $password = $pwd
  22       } else {
  23           if (Test-Path $tokenfile) {
  24             Get-Content $tokenfile | Foreach-Object{
  25                $var = $_.Split('=')
  26                Set-Variable -Name $var[0] -Value $var[1]
  27                 }
  28           } else {
  29                 Write-host "The user_token file $tokenfile was not found" -foregroundcolor "magenta"
  30                 return -1 }
  31         
  32         
  33       }
  34     Write-host "Skytap user is $username"
  35     $auth = [Convert]::ToBase64String([Text.Encoding]::ASCII.GetBytes(("{0}:{1}" -f $username,$password)))
  36     $global:headers = @{"Accept" = "application/json"; Authorization=("Basic {0}" -f $auth)}
  37     return 0
  38 }

  45 function Show-RequestFailure  {
  46     $ex = $global:errorResponse
  47     if ($ex.gettype().fullname -eq 'System.Net.WebException') {
  48         $nob = New-Object -TypeName psobject -Property @{
  49         requestResultCode = [int]$ex.HResult
  50         eDescription = $ex.gettype()
  51         eMessage = $ex.Message
  52         method = $ex.Source
  53         }
  54         
  55        }else{
  56         $eresp = $ex.response 
  57         $errorResponse = $eresp.GetResponseStream()
  58         $reader = New-Object System.IO.StreamReader($errorResponse)
  59         $reader.BaseStream.Position = 0
  60         $reader.DiscardBufferedData()
  61         $responseBody = $reader.ReadToEnd();
  62         $nob = New-Object -TypeName psobject -Property @{
  63             requestResultCode = [int]$eresp.StatusCode
  64             eDescription = $eresp.StatusDescription
  65             eMessage = $responseBody
  66             method = $eresp.Method
  67         }
  68     }
  69     $global:errorResponse = ''
  70     return $nob
  71 }

 732 function Get-Templates ([string]$templateId, [string]$attributes,[string]$v2='T',[int]$startCount="100",[string]$qscope='company') {
 733 <#
 734     .SYNOPSIS
 735       Get template(s) optionally filter by attributes
 736     .SYNTAX
 737       Get-Templates [templateId] [attribute value pairs]
 738        Returns template(s) list object
 739     .EXAMPLE
 740           All templates
 741             Get-Templates
 742             
 743          Only templates that are non-public in US-West
 744             Get-Templates -attributes "public=False,region=USWest"
 745             
 746         Only specific template details
 747             Get-Templates 12345
 748   #>
 749         $more_records = $True
 750         if ($v2 -eq 'T') {
 751                 While ($more_records) {
 752                     try {
 753                         if ($attributes){
 754                             $uri = $global:url + '/v2/templates?scope=' + $qscope + '&count=' + $startCount + '&offset=' + $global:tOffset + '&query=' + $attributes
 755                         }else{
 756                             if ($templateId){   
 757                                 $uri = $global:url + '/v2/templates/' + $templateId
 758                             }else{
 759                                 $uri = $global:url + '/v2/templates?scope=' + $qscope + '&count=' + $startCount + '&offset=' + $global:tOffset
 760                                 }
 761                             }
 762                         write-host $uri
 763                         $result = Invoke-WebRequest -Uri $uri -Method GET -ContentType 'application/json' -Headers $global:headers 
 764                                         
 765                             } catch { 
 766                                 $global:errorResponse = $_.Exception
 767                                 $result = Show-RequestFailure       
 768                             }
 769                         if ($result.StatusCode -ne 200) {
 770                             write-host $result.StatusCode
 771                             write-host $result.StatusDescription
 772                                     return
 773                                     }
 774                         $hold_result = $hold_result + (ConvertFrom-Json $result.Content)
 775     
 776                         $hdr = $result.headers['Content-Range']
 777                         
 778                         if ($hdr.length -gt 0) {
 779                             $hcounters = $hdr.Split('-')[1]
 780                             [Int]$lastItem,[int]$itemTotal = $hcounters.Split('/')
 781                             
 782                             if (($lastItem + 1)  -lt ($itemTotal)){                                         
 783                                 $global:tOffset = $lastItem + 1
 784                             }
 785                             else 
 786                             {
 787                                 $more_records = $False
 788                             }
 789                         }
 790                         else 
 791                         {
 792                             $more_records = $False
 793                         }
 794                     }
 795                     write-host $hold_result.count
 796                     $result =  $hold_result
 797                     $result | Add-member -MemberType NoteProperty -name requestResultCode -value 0
 798                     $global:tOffset = 0
 799                     return $result
 800                     
 801                 } else {
 802                     try {
 803                         if ($templateId){   
 804                             $uri = "$global:url/templates/$templateId"
 805                         }else{
 806                             $uri = "$global:url/templates"
 807                             }
 808                     $result = Invoke-RestMethod -Uri $uri -Method GET -ContentType "application/json" -Headers $global:headers 
 809                     $result | Add-member -MemberType NoteProperty -name requestResultCode -value 0
 810                         } catch {

 814                     return $result
